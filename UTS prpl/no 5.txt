    function inputValidation($email, $password)     // function untuk membuat input data dan memvalidasi inputan user

    {

        if (empty($email) || empty($password)) {       // untuk mengecek apabila inputan masih kosong

            return 'Username or password is empty!';    // apabila ada yang kosong akan keluar peringatan empty nya

        }

        if (!filter_var($email, FILTER_VALIDATE_EMAIL)) {     // untuk validasi penamaan email, seperti @ dan .com

            return "Invalid email format!";                    // untuk peringatan email ada yang tidak valid

        }

        if (strlen($password) < 6) {                        // password minimal 6 karakter

            return "Your Password Must Contain At Least 6 Characters!"; // peringatan apabila password kurang dari 6 karakter

        }

        if (!preg_match("#[0-9]+#", $password) && !preg_match("#[A-Z]+#", $password)) { // password minimal 1 karakter kapital dan numerik

            return "Your Password Must Contain At Least 1 Number or capital letter! "; // peringatan apabila password belum ada 1 karakter kapital dan numerik

        }

       

        session_start();

        $_SESSION['username'] = $email;         // akan mengambil username dari email yang diinputkan

        header("location: dashboard.php");      // apabila sudah maka akan menuju ke halaman dashboard

      }

}
